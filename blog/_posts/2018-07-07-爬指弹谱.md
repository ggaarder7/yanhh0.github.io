神经网络（笑）。注意到文章内的图片有一个特别的属性data-src，似乎指向
图片地址。

def get_img(lnk):
    wd.get(lnk)
    soup = BeautifulSoup(wd.page_source, 'html5lib')
    out = []

    for img in soup.find_all('img'):
        if 'data-src' in img.attrs:
            out.append(img['data-src'])

    return out
